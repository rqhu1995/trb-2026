\begin{block}{10. Cost-Effectiveness of Repairers}
    \textbf{Indicator $\nu$:} Ratio of dissatisfaction reduction to repairer wage
    \[
    \nu = \frac{c^p \sum_{i \in S}(\widehat{F}_i - \widehat{F}'_i)}{\varpi \cdot T / 3600}
    \]
    $\nu > 1$: Cost-effective \hspace{1cm} $\nu < 1$: Not cost-effective
