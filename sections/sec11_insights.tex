\begin{block}{11. Key Managerial Insights}
    \begin{itemize}
        \item \textcolor{AlertRed}{\textbf{Critical Threshold:}} Repair time $>$ 645s $\Rightarrow$ \textbf{never} cost-effective
        \item \textbf{Low damage ($\ell < 20\%$):} Trucks alone sufficient; repairers not justified
        \item \textbf{High damage ($\ell \geq 30\%$):} On-site repairers become economically justified
        \item \textbf{Preventive maintenance} crucial to keep repair times below threshold
    \end{itemize}
\end{block}
