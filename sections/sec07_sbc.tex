\begin{block}{7. Station Budget Concept (SBC)}
    \textbf{Key Innovation:} Prevents over-allocation of service time to early stations
    
    \vspace{0.3cm}
    \textbf{Benefit-Cost Ratio Function (BCRF):}
    \[
    BCRF_s^t(p,b) = \frac{F_s(p,b) - F_s(q,0)}{t^{load} \cdot (|q-p|+b)}
    \]
    
    \begin{itemize}
        \item Quantifies dissatisfaction reduction per unit service time
        \item Higher BCRF stations receive larger time budgets
        \item Unused time redistributed dynamically to later stations
    \end{itemize}
\end{block}
