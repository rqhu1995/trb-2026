    \centering
    \textbf{Key Decision Rules:}
    \begin{itemize}
        \item \textbf{< 20\% broken:} Trucks alone sufficient
        \item \textbf{> 30\% broken:} On-site repairers cost-effective
        \item \textbf{Repair time > 11 min:} Never cost-effective
    \end{itemize}
