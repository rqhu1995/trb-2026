\begin{block}{2. Contributions}
    \begin{enumerate}
        \item \textbf{Novel Problem:} First study to integrate on-site repairs with vehicle-based repositioning operations
        \item \textbf{MILP Model:} Time-indexed formulation minimizing user dissatisfaction + CO$_2$ emissions
        \item \textbf{HGSDAC-SBC:} Hybrid Genetic Search algorithm with Station Budget Constraint for large-scale instances
        \item \textbf{Managerial Insights:} Cost-effectiveness analysis framework for repairer deployment decisions
    \end{enumerate}
\end{block}
